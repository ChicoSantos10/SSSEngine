126 template<MatrixTypeConcept T, MatrixTypeConcept V>
127 requires(std::same_as<T, V>) && (T::Columns() == V::Rows())
128 SSSENGINE_GLOBAL constexpr auto operator*(const T &lhs, const V &rhs)
129 {
130 using Type = typename T::Type;
131 constexpr MatrixSize RowsLhs = T::Rows();
132 constexpr MatrixSize RowsRhs = V::Rows();
133 constexpr MatrixSize ColumnsRhs = V::Columns();
134
135 Matrix<Type, ColumnsRhs, RowsRhs> result;
136
137 for(MatrixSize i = 0; i < RowsLhs; ++i)
138 {
139 for(MatrixSize k = 0; k < RowsLhs; ++k)
140 {
141 for(MatrixSize j = 0; j < RowsLhs; ++j)
142 {
143 result[i, j] += lhs[i, k] * rhs[k, j];
144 }
145 }
146 }
147
148 return result;
149 }
